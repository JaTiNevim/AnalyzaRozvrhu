229  private static void Handle_Roak(STAG_Classes.STAG_Database data, STAG_Classes.Student student, STAG_Classes.RozvrhovaAkce roak, JsonSerializer serializer)
230  {
231  // Zajistím aby mi nechyběl předmět této rozvrhové akce
232  // Zkontroluji, jestli už znám katedru toho předmětu
233  if (!data.PredmetyPodleKateder.ContainsKey(roak.Katedra))
234  {
235  // Zatim neznam ~ neni z me fakultu... musim stahnout (nebyl obsažen v predmetyByFakultaFullInfo)
236  Handle_PredmetInfo(roak.Katedra, roak.Predmet, data, serializer);
237  }
238  // Zkontroluji, jestli už ten předmět znám
239  if (!data.PredmetyPodleKateder[roak.Katedra].ContainsKey(roak.Predmet))
240  {
241  // Nezman predmet... musim stahnout
242  Handle_PredmetInfo(roak.Katedra, roak.Predmet, data, serializer);
243  }
244 
245  // vvvvv REFERENCE vvvvvv
246 
247  // Ulozim si akci a ridam reference student => roak
248  if (!data.Akce.ContainsKey(roak.RoakIdno))
249  {
250  data.Akce.Add(roak.RoakIdno, roak);
251 
252  // Pridam reference roak <=> predmet
253  data.PredmetyPodleKateder[roak.Katedra][roak.Predmet].VsechnyAkce.Add(roak);
254  roak.PredmetRef = data.PredmetyPodleKateder[roak.Katedra][roak.Predmet];
255  }
256 
257  // Přidám akci do rozvrhu studenta
258  student.Rozvrh.Add(data.Akce[roak.RoakIdno]);
259  // Pro zajímavost zjišťuji kolik studentů skutečně ukazuje na tuto akci (porovnání obsazenosti vs. referencí)
260  data.Akce[roak.RoakIdno].referenceCount++;
